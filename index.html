Executive Summary

1. Core Parameters & Architecture

Key System Parameters

  • Blockchain: Binance Smart Chain (BSC)
  • Block Time: 1.5 seconds (57,600 blocks/day)
  • Halving Frequency: Every 4,204,800 blocks (73 days)
  • Hash Rate Formula: 1 RICE = 0.1 H/s
  • Burn Mechanism: Staked tokens sent to burn address
  • Reward Distribution: Proportional to staked amount

BPOS Mechanism

Staking Process: 1. User stakes RICE tokens 2. Tokens automatically sent to burn address (0x...dEaD) 3. User receives hashrate: Staked RICE × 0.1 H/s 4. Daily rewards calculated based on network share 5. Rewards distributed proportionally to hashrate Reward Formula: Daily Reward = (User Hashrate ÷ Total Network Hashrate) × Daily Emission

4. Smart Contract Implementation

pragma solidity ^0.8.19; contract BalancedDescendingBPOS { uint256 public constant TOTAL_SUPPLY = 446400 * 1e18; uint256 public constant BLOCKS_PER_HALVING = 4204800; // 73 days on BSC uint256 public constant HASHRATE_MULTIPLIER = 1e17; // 1 RICE = 0.1 H/s address public constant BURN_ADDRESS = 0x000000000000000000000000000000000000dEaD; uint256 public startTime; uint256 public totalStaked; uint256 public totalMined; // Balanced descending emission schedule (daily RICE amounts) uint256[30] public dailyEmissions = [ // Year 1: Moderate start (18.8%) 250, 240, 230, 220, 210, // Year 2: Building momentum (22.9%) 260, 270, 280, 290, 300, // Year 3: Peak rewards (27.0%) 310, 320, 330, 340, 350, // Year 4: Sustained high (21.3%) 320, 290, 260, 230, 200, // Year 5: Gradual decline (9.0%) 170, 140, 110, 80, 50, // Year 6: Final phase (1.0%) 20, 15, 10, 8, 6 ]; struct StakeInfo { uint256 amount; // RICE staked (burned) uint256 hashrate; // Calculated hashrate uint256 lastReward; // Last block rewarded uint256 stakeTime; // Timestamp of stake } mapping(address => StakeInfo) public stakes; function getCurrentDailyEmission() public view returns (uint256) { uint256 daysPassed = (block.timestamp - startTime) / 1 days; uint256 currentHalving = daysPassed / 73; if (currentHalving >= 30) return 0; return dailyEmissions[currentHalving] * 1e18; } function getCurrentEmissionPerBlock() public view returns (uint256) { return getCurrentDailyEmission() / 57600; // BSC blocks per day } function stakeRice(uint256 amount) external { require(amount > 0, "Cannot stake 0"); require(totalMined < TOTAL_SUPPLY, "Mining complete"); // Transfer RICE to burn address (staking = burning) IERC20(riceToken).transferFrom(msg.sender, BURN_ADDRESS, amount); // Update stake info stakes[msg.sender].amount += amount; stakes[msg.sender].hashrate = stakes[msg.sender].amount * HASHRATE_MULTIPLIER / 1e18; stakes[msg.sender].lastReward = block.number; stakes[msg.sender].stakeTime = block.timestamp; totalStaked += amount; emit Staked(msg.sender, amount, stakes[msg.sender].hashrate); } function calculateReward(address user) public view returns (uint256) { StakeInfo memory stake = stakes[user]; if (stake.amount == 0 || totalStaked == 0) return 0; uint256 blocksToReward = block.number - stake.lastReward; uint256 userShare = stake.hashrate * 1e18 / getTotalHashrate(); uint256 totalReward = 0; for (uint256 i = 0; i < blocksToReward; i++) { uint256 blockNum = stake.lastReward + i + 1; uint256 emissionAtBlock = getEmissionAtBlock(blockNum); totalReward += emissionAtBlock * userShare / 1e18; } return totalReward; } function claimRewards() external { uint256 reward = calculateReward(msg.sender); require(reward > 0, "No rewards available"); stakes[msg.sender].lastReward = block.number; totalMined += reward; IERC20(riceToken).mint(msg.sender, reward); emit RewardClaimed(msg.sender, reward); } function getTotalHashrate() public view returns (uint256) { return totalStaked * HASHRATE_MULTIPLIER / 1e18; } function getEmissionPhase() public view returns (string memory) { uint256 daysPassed = (block.timestamp - startTime) / 1 days; if (daysPassed <= 365) return "Moderate Start"; if (daysPassed <= 730) return "Building Momentum"; if (daysPassed <= 1095) return "Peak Rewards"; if (daysPassed <= 1460) return "Sustained High"; if (daysPassed <= 1825) return "Gradual Decline"; return "Final Phase"; } event Staked(address indexed user, uint256 amount, uint256 hashrate); event RewardClaimed(address indexed user, uint256 reward); }
